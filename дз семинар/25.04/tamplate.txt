#include <iostream>

using namespace std;

template<typename T>
T* aarray(int n) 
{
    T* arr = new T[n];
    for (int i = 0; i < n; i++)
        cin >> arr[i];
    for (int i = 0; i < n; i++)
        cout << arr[i] << " ";
        cout << arr[i] << " ";
    cout << endl;
    return arr;
}

int main() {
    int* arr = aarray<int>(10);
return 0;
